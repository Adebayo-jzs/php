/* 1. Make parent relative */
.box { position: relative; padding: 24px; }

/* 2. Place corner brackets with pseudos */
.box::before, .box::after,
.box .corners::before, .box .corners::after {
  content: '';
  position: absolute;
  width: 16px;
  height: 16px;
  border: 2px solid white;
}

/* 3. Clip each corner's visible sides */
.box::before  { top:0; left:0;  border-width: 2px 0 0 2px; } /* ┌ */
.box::after   { top:0; right:0; border-width: 2px 2px 0 0; } /* ┐ */
.corners::before { bottom:0; left:0;  border-width: 0 0 2px 2px; } /* └ */
.corners::after  { bottom:0; right:0; border-width: 0 2px 2px 0; } /* ┘ */

/* 4. Hover: corners grow inward */
.box:hover::before,
.box:hover .corners::before { width: 100%; height: 100%; }
